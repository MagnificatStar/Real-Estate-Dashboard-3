FUNCTION FnMoMGrowthPct= (metric:INT64 EXPR) =>
VAR _CurrentMonth= metric
VAR _LastMonth = 
    CALCULATE(
        metric,
        DATEADD('Date'[Date], -1,MONTH)
    )
    VAR _VarPct = 
    DIVIDE(
        (_CurrentMonth - _LastMonth),
        _LastMonth,
        0
    )
    RETURN 
    IF(_VarPct > 0, "↗", "↘")
    & " "
    &FORMAT(_VarPct, "0%")
    & " vs last month"

    EVALUATE
    ROW(
        "MoM Test Result", FnMoMGrowthPct([Active listings])
    )
